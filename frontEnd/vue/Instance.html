# filter

转换数据 transform

/**
*
* val 页面中传过来的只
*/
Vue.filter('changeTableColumns',function(val,params){
    let result = '异常'
    if(val){
        result = '正常'
    }
    return result
})

// default
<div>{{val | changeTableColumns}}</div>
//从第二个参数开始 传递 自定义参数
<div>{{val | changeTableColumns(val,'custom-params')}}</div>
